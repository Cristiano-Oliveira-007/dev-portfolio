# Diferença de "saudacao" para "apresentacao" é:
     Variavél usada em saudacao é "let nome = 'Bruno'" que foi declarada no
Escopo GLOBAL.
    já a variavél "let nome = 'Saldanha'" foi declarada dentro de "apresentacao"
ou seja a variavél de "apresentacao" é uma variavél "PRIVADA"

- function apresentandoAlguem(){
    let nome = "Cristiano"
    function saudacao(){
        console.log('olá, ${nome}')
    }
    function despedida(){
        console.log('Até mais, ${nome}')
    }
    return{
        saudacao: saudacao,
        despedida: despedida
    }
}

- const pessoa = apresentandoAlguem()
- pessoa.saudacao()
Olá, Cristiano
- pessoa.despedida
Até mais, Cristiano
